*  내가 푼 코드
```
class Solution {
    fun solution(n: Long): Long {
        var answer: Long = 0

        do {
            answer++
            if(answer * answer == n) {
                return (answer + 1) * (answer + 1) 
            }
        } while(answer <= n)

        return -1
    }
}
```
